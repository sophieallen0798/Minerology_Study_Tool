package model.activities;
import model.Learned;
import model.Mineral;
import model.Review;
import ui.MinApp;

import java.util.ArrayList;
import java.util.List;
import java.util.Random;
import java.util.Scanner;

public class Study extends MinApp {
    protected int mineralsStudied;
    private int currentScore;
    private Scanner input;

    public Study() {
        currentScore = 0;
        mineralsStudied = 0;
        System.out.println("Which lab(s) do you want to study? Enter lab numbers separated by commas.");
        input = new Scanner(System.in);
    }

    // EFFECTS: user select next display property or choose to guess
    private void studyMenu() {
        mineralsStudied += 1;
        System.out.println("Mineral Number" + mineralsStudied);
        System.out.println("Select a property to view.");
        System.out.println("c for color" + "h for hardness" + "cs for crystal system" + "or g to guess mineral name");
    }

    // EFFECTS: keeps generating new mineral and study options
    private void game() {
        Mineral currentMin = selectNextStudyMineral();
        studyMenu();
        input = new Scanner(System.in);
        if (input.next() == "g") {
            input = new Scanner(System.in);
            if (checkGuess(input.next(), currentMin)) {
                promptToMove(currentMin);
            }
        }
    }

    //EFFECTS: check if use guessed mineral correctly
    private boolean checkGuess(String guess, Mineral currentMineral) {
        if (currentMineral.getName() == guess) {
            System.out.println("Correct! +1 point");
            currentScore += 1;
            return true;
        }
        System.out.println("Incorrect, the mineral was" + currentMineral.getName());
        return false;
    }

    // EFFECTS: ask user to move mineral then resume game
    private void promptToMove(Mineral currentMineral) {
        System.out.println("Move" + currentMineral.getName() + "to learned? y for yes, n for no");
        input = new Scanner(System.in);
        String inputString = input.next();
        if (inputString == "y") {
            moveToLearned(currentMineral);
            game();
        } else if (inputString == "n") {
            game();
        } else {
            System.out.println("Please enter y for yes or n for no");
            promptToMove(currentMineral);
        }
    }

    //EFFECTS: select random next mineral to study from review list
    private Mineral selectNextStudyMineral() {
        Random rand = new Random();
        List<Mineral> reviewList = new ArrayList<>();
        reviewList = toReview.getReviewList();
        Mineral selectedMineral = reviewList.get(rand.nextInt(reviewList.size()));
        return selectedMineral;
    }

}
