package model.activities;

import model.Mineral;
import model.Review;
import ui.MinApp;

import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

public class AddMinerals extends MinApp {
    private Scanner input;
    private static int NUM_PROPERTIES = 5;

    public AddMinerals() {
        addMineralPrompts();
    }

    // EFFECTS: take mineral properties as inputs and create mineral
    // add mineral to the Review list
    private Mineral addMineralPrompts() {
        Mineral nextMin = new Mineral();
        //System.out.println("What would you like to do?");
        int i = 0;
        while (i < NUM_PROPERTIES) {
            System.out.println(messagesList(i));
            input = new Scanner(System.in);
            setProperties(nextMin, input, i);
            i += 1;
        }
        printMineral(nextMin);
        toReview.addToReview(nextMin);
        return (nextMin);
    }

    // EFFECTS: return requested prompt message by index int
    public String messagesList(int i) {
        List<String> messages = new ArrayList<>();
        messages.add("\nLab Number:");
        messages.add("\nMineral Name:");
        messages.add("\nColor:");
        messages.add("\nHardness:");
        messages.add("\nCrystal System: "
                + "\ni for isometric, "
                + "tetra for tetragonal, "
                + "tri for triclinic, "
                + "h for hexagonal, "
                + "m for monoclinic,  "
                + "o for orthorhombic");
        return messages.get(i);
    }

    // EFFECTS: print mineral and it's properties
    public void printMineral(Mineral mineralToPrint) {
        System.out.println("Lab: " + mineralToPrint.getLab());
        System.out.println("Name: " + mineralToPrint.getName());
        System.out.println("Color: " + mineralToPrint.getColor());
        System.out.println("Hardness: " + mineralToPrint.getHardness());
        System.out.println("Crystal System: " + mineralToPrint.getCrystalSystem());
        // get it to print full name
        //System.out.println("Crystal System: " + crystalSystemName(mineralToPrint.getCrystalSystem()));
    }

    // MODIFIES: this
    // EFFECTS: modifies specified field of the given mineral
    public void setProperties(Mineral m, Scanner inpu, int i) {
        if (i == 0) {
            m.setLab(inpu.nextInt());
        } else if (i == 1) {
            m.setName(inpu.next());
        } else if (i == 2) {
            m.setColor(inpu.next());
        } else if (i == 3) {
            m.setHardness(inpu.nextInt());
        } else if (i == 4) {
            m.setCrystalSystem(inpu.next().toLowerCase());
        }
    }
}
